Routing

How URLs map to controllers

// Program.cs - Configure routing
var builder = WebApplication.CreateBuilder(args);
var app = builder.Build();

app.UseRouting();

app.MapControllerRoute(
    name: "default",
    pattern: "{controller=Home}/{action=Index}/{id?}");

app.Run();

// URL examples:
// /                 -> HomeController.Index
// /Home/Privacy    -> HomeController.Privacy
// /Products/5      -> ProductsController.Details(5)

URL pattern matched to controller/action names.

Attributes directly on controller methods.

Working with Forms

GET and POST requests

public class ContactController : Controller
{
    // GET: /Contact
    [HttpGet]
    public IActionResult Index()
    {
        return View();
    }

    // POST: /Contact
    [HttpPost]
    public IActionResult Index(ContactForm model)
    {
        if (.ModelState.IsValid)
        {
            // Process the form data
            _service.SaveContact(model);
            return RedirectToAction("Success");
        }
        return View(model);
    }
}
💡 ModelState.IsValid

Always check ModelState.IsValid before processing form data. It validates all [Required], [MaxLength], and other data annotations.

CRUD Operations

Create, Read, Update, Delete

using var context = new AppDbContext();

// CREATE - Add new record
var user = new User { Name = "John", Email = "john@example.com" };
context.Users.Add(user);
context.SaveChanges();

// READ - Query records
var allUsers = context.Users.ToList();
var john = context.Users.FirstOrDefault(u => u.Name == "John");

// UPDATE - Modify record
john.Email = "newemail@example.com";
context.SaveChanges();

// DELETE - Remove record
context.Users.Remove(john);
context.SaveChanges();

Creating a Web API

flowchart LR subgraph api["REST API Flow"] Req["📨 HTTP Request"] --> C["📝 API Controller"] C --> S["📦 Service Layer"] S --> Db["🗄️ Database"] Db --> Json["📄 JSON Response"] end style Req fill:#58a6ff,color:#fff style C fill:#a371f7,color:#fff style S fill:#3fb950,color:#fff style Db fill:#d29922,color:#fff style Json fill:#f85149,color:#fff 
// Controllers/ProductsController.cs
[ApiController]
[Route("api/[controller]")]
public class ProductsController : ControllerBase
{
    private readonly AppDbContext _context;

    public ProductsController(AppDbContext context)
    {
        _context = context;
    }

    // GET: api/products
    [HttpGet]
    public ActionResult<List<Product>> GetProducts()
    {
        return _context.Products.ToList();
    }

    // GET: api/products/5
    [HttpGet("{id}")]
    public ActionResult<Product> GetProduct(int id)
    {
        var product = _context.Products.Find(id);
        if (product == null)
            return NotFound();
        return product;
    }
}

CRUD Operations in API

POST, PUT, DELETE

// POST: api/products
[HttpPost]
public ActionResult<Product> PostProduct(Product product)
{
    _context.Products.Add(product);
    _context.SaveChanges();
    return CreatedAtAction("GetProduct", new { id = product.Id }, product);
}

// PUT: api/products/5
[HttpPut("{id}")]
public ActionResult<Product> PutProduct(int id, Product product)
{
    if (id != product.Id)
        return BadRequest();

    _context.Entry(product).State = EntityState.Modified;
    _context.SaveChanges();
    return NoContent();
}

// DELETE: api/products/5
[HttpDelete("{id}")]
public ActionResult<Product> DeleteProduct(int id)
{
    var product = _context.Products.Find(id);
    if (product == null)
        return NotFound();

    _context.Products.Remove(product);
    _context.SaveChanges();
    return NoContent();
}

JWT Authentication

Token-based auth for APIs

// Install JWT package
dotnet add package Microsoft.AspNetCore.Authentication.JwtBearer

// Program.cs - Configure JWT
builder.Services.AddAuthentication(JwtBearerDefaults.AuthenticationScheme)
    .AddJwtBearer(options =>
    {
        options.TokenValidationParameters = new TokenValidationParameters
        {
            ValidateIssuer = true,
            ValidateAudience = true,
            ValidateLifetime = true,
            ValidIssuer = "myapp",
            ValidAudience = "myapp",
            IssuerSigningKey = new SymmetricSecurityKey(
                Encoding.UTF8.GetBytes("your-secret-key-here"))
        };
    });
// Generate JWT token
public string GenerateJwtToken(IdentityUser user)
{
    var claims = new List<Claim>
    {
        new Claim(ClaimTypes.NameIdentifier, user.Id),
        new Claim(ClaimTypes.Name, user.Email)
    };

    var key = new SymmetricSecurityKey(
        Encoding.UTF8.GetBytes("your-secret-key"));
    var creds = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);

    var token = new JwtSecurityToken(
        issuer: "myapp",
        audience: "myapp",
        claims: claims,
        expires: DateTime.Now().AddDays(7),
        signingCredentials: creds);

    return new JwtSecurityTokenHandler().WriteToken(token);
}

Authorization

Protect your controllers

// Require authenticated user
[Authorize]
public class SecureController : ControllerBase
{
    // Require specific role
    [Authorize(Roles = "Admin")]
    [HttpGet("admin")]
    public IActionResult AdminOnly()
    {
        return Ok("Admin area");
    }

    // Require policy
    [Authorize(Policy = "RequireDeveloper")]
    [HttpGet("developer")]
    public IActionResult DeveloperOnly()
    {
        return Ok("Developer area");
    }
}
💡 Authorization vs Authentication

Authentication = Who are you? (Login)
Authorization = What can you do? (Roles/Policies)

Writing Unit Tests

Test your business logic

public class CalculatorTests
{
    [Fact]
    public void Add_ReturnsSum()
    {
        // Arrange
        var calculator = new Calculator();

        // Act
        var result = calculator.Add(2, 3);

        // Assert
        Assert.Equal(5, result);
    }

    [Theory]
    [InlineData(1, 1, 2)]
    [InlineData(2, 3, 5)]
    [InlineData(10, 20, 30)]
    public void Add_ReturnsCorrectSum(int a, int b, int expected)
    {
        var calculator = new Calculator();
        var result = calculator.Add(a, b);
        Assert.Equal(expected, result);
    }
}

Integration Tests

Test your API end-to-end

// Install integration testing package
dotnet add package Microsoft.AspNetCore.Mvc.Testing

public class ProductApiTests : IClassFixture<WebApplicationFactory<Program>>
{
    private readonly HttpClient _client;

    public ProductApiTests(WebApplicationFactory<Program> factory)
    {
        _client = factory.CreateClient();
    }

    [Fact]
    public async Task GetProducts_ReturnsSuccess()
    {
        // Act
        var response = await _client.GetAsync("/api/products");

        // Assert
        response.EnsureSuccessStatusCode();
        var content = await response.Content.ReadAsStringAsync();
        Assert.NotEmpty(content);
    }
}

Aim for 70-80% code coverage on your business logic. Focus on testing edge cases and error handling, not trivial code.
